SELECT p.`product_id`, p.`product_name`, p.`description`, c.`category_name`, p.`barcode`, p.`product_category_id`,
l.`current_stock_level`
FROM products p
INNER JOIN product_location_categories c ON c.`category_id` = p.`product_category_id`
INNER JOIN product_stock_levels l ON l.`product_id` = p.`product_id`
ORDER BY p.`product_name`;


SELECT p.`product_id`, p.`product_name`, p.`description`, p.`barcode`, c.`category_name`, p.`price`, p.`product_category_id`,
u.`unit_abbreviation`, p.`weight`
FROM products p
INNER JOIN product_location_categories c ON c.`category_id` = p.`product_category_id`
LEFT JOIN measurement_units u ON u.`unit_id` = p.`unit_id`
WHERE p.`product_id` = 1
